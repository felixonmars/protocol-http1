def read
	return nil if @finished
	
	# It is possible this line contains chunk extension, so we use `to_i` to only consider the initial integral part:
	length = read_line.to_i(16)
	
	if length == 0
		@finished = true
		
		read_trailer
		
		return nil
	end
	
	# Read trailing CRLF:
	chunk = @stream.read(length + 2)
	
	# ...and chomp it off:
	chunk.chomp!(CRLF)
	
	@length += length
	@count += 1
	
	return chunk
end
